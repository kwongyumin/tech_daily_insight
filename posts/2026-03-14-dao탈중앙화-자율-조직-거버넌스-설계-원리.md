# DAO(탈중앙화 자율 조직) 거버넌스 설계 원리

## 개요

블록체인 기술이 성숙하면서 DAO(Decentralized Autonomous Organization)는 단순한 개념 실험을 넘어 실제 수십억 달러 규모의 프로토콜을 운영하는 핵심 메커니즘으로 자리 잡았다. Compound, Uniswap, MakerDAO 같은 DeFi 프로토콜은 모두 DAO 거버넌스를 통해 의사결정을 내린다.

그러나 많은 개발자들이 DAO를 단순히 "투표 컨트랙트"로 이해하는 경향이 있다. 실제로는 **토큰 경제학**, **시빌 어택 방어**, **타임락**, **위임 메커니즘**, **쿼럼 설계** 등 복잡한 시스템 엔지니어링 문제가 얽혀 있다. 이 포스팅에서는 DAO 거버넌스의 핵심 설계 원리를 실제 Solidity 코드와 함께 살펴본다.

---

## 핵심 개념

### 1. 거버넌스 토큰과 투표권

DAO의 가장 기본적인 구성 요소는 거버넌스 토큰이다. ERC-20 토큰에 투표 기능을 추가한 **ERC-20Votes** 표준이 사실상 업계 표준으로 자리 잡았다.

핵심 메커니즘:
- **체크포인트(Checkpoint)**: 특정 블록 번호 시점의 토큰 잔액을 기록
- **위임(Delegation)**: 토큰 보유자가 다른 주소에 투표권을 위임
- **스냅샷**: 제안 생성 시점의 투표권을 고정하여 플래시론 공격 방지

### 2. 거버넌스 생명주기

```
제안(Proposal) → 투표 지연(Voting Delay) → 투표 기간(Voting Period) → 
실행 대기(Timelock Queue) → 실행(Execute) 또는 취소(Cancel)
```

각 단계의 파라미터 설정은 보안과 민첩성 사이의 트레이드오프다.

### 3. 쿼럼(Quorum) 설계

쿼럼은 제안이 통과되기 위한 최소 투표 참여 요건이다. 고정 쿼럼과 동적 쿼럼 두 가지 방식이 있다:

- **고정 쿼럼**: 전체 공급량의 N% 이상 참여 필요. 단순하지만 토큰 공급량 변화에 취약
- **동적 쿼럼**: 최근 투표 참여율이나 제안의 위험도에 따라 쿼럼 조정 (MakerDAO 방식)

## 실전 예제

### 거버넌스 토큰 구현

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/extensions/ERC20Votes.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract GovernanceToken is ERC20Votes, Ownable {
    uint256 public constant MAX_SUPPLY = 100_000_000 * 1e18; // 1억 토큰

    constructor(address initialOwner)
        ERC20("MyDAO Token", "MDAO")
        EIP712("MyDAO Token", "1")
        Ownable(initialOwner)
    {
        // 초기 공급량의 30%를 트레저리에 할당
        _mint(initialOwner, MAX_SUPPLY * 30 / 100);
    }

    function mint(address to, uint256 amount) external onlyOwner {
        require(totalSupply() + amount <= MAX_SUPPLY, "Exceeds max supply");
        _mint(to, amount);
    }

    // 투표권은 반드시 위임해야 활성화됨
    // 자기 자신에게 위임하면 직접 투표 가능
    function _update(
        address from,
        address to,
        uint256 value
    ) internal override(ERC20Votes) {
        super._update(from, to, value);
    }
}
```

### 핵심 거버넌스 컨트랙트

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/governance/Governor.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorSettings.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorCountingSimple.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorVotes.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorVotesQuorumFraction.sol";
import "@openzeppelin/contracts/governance/extensions/GovernorTimelockControl.sol";

contract MyDAOGovernor is
    Governor,
    GovernorSettings,
    GovernorCountingSimple,
    GovernorVotes,
    GovernorVotesQuorumFraction,
    GovernorTimelockControl
{
    constructor(
        IVotes _token,
        TimelockController _timelock
    )
        Governor("MyDAOGovernor")
        GovernorSettings(
            1 days / 12,    // votingDelay: 약 1일 (12초/블록 기준 7200블록)
            5 days / 12,    // votingPeriod: 약 5일
            100_000 * 1e18  // proposalThreshold: 10만 토큰 보유해야 제안 가능
        )
        GovernorVotes(_token)
        GovernorVotesQuorumFraction(4) // 전체 공급량의 4% 이상 참여 필요
        GovernorTimelockControl(_timelock)
    {}

    // 필수 오버라이드
    function votingDelay()
        public view override(Governor, GovernorSettings)
        returns (uint256)
    {
        return super.votingDelay();
    }

    function votingPeriod()
        public view override(Governor, GovernorSettings)
        returns (uint256)
    {
        return super.votingPeriod();
    }

    function quorum(uint256 blockNumber)
        public view override(Governor, GovernorVotesQuorumFraction)
        returns (uint256)
    {
        return super.quorum(blockNumber);
    }

    function state(uint256 proposalId)
        public view override(Governor, GovernorTimelockControl)
        returns (ProposalState)
    {
        return super.state(proposalId);
    }

    function proposalNeedsQueuing(uint256 proposalId)
        public view override(Governor, GovernorTimelockControl)
        returns (bool)
    {
        return super.proposalNeedsQueuing(proposalId);
    }

    function proposalThreshold()
        public view override(Governor, GovernorSettings)
        returns (uint256)
    {
        return super.proposalThreshold();
    }

    function _queueOperations(
        uint256 proposalId,
        address[] memory targets,
        uint256[] memory values,
        bytes[] memory calldatas,
        bytes32 descriptionHash
    ) internal override(Governor, GovernorTimelockControl) returns (uint48) {
        return super._queueOperations(
            proposalId, targets, values, calldatas, descriptionHash
        );
    }

    function _executeOperations(
        uint256 proposalId,
        address[] memory targets,
        uint256[] memory values,
        bytes[] memory calldatas,
        bytes32 descriptionHash
    ) internal override(Governor, GovernorTimelockControl) {
        super._executeOperations(
            proposalId, targets, values, calldatas, descriptionHash
        );
    }

    function _cancel(
        address[] memory targets,
        uint256[] memory values,
        bytes[] memory calldatas,
        bytes32 descriptionHash
    ) internal override(Governor, GovernorTimelockControl) returns (uint256) {
        return super._cancel(targets, values, calldatas, descriptionHash);
    }

    function _executor()
        internal view override(Governor, GovernorTimelockControl)
        returns (address)
    {
        return super._executor();
    }
}
```

### 타임락 컨트랙트 설정

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/governance/TimelockController.sol";

// 배포 스크립트 (Foundry 기준)
contract DeployDAO {
    function deploy() external returns (
        GovernanceToken token,
        TimelockController timelock,
        MyDAOGovernor governor
    ) {
        address deployer = msg.sender;
        
        // 1. 거버넌스 토큰 배포
        token = new GovernanceToken(deployer);
        
        // 2. 타임락 배포 (최소 2일 지연)
        address[] memory proposers = new address[](1);
        address[] memory executors = new address[](1);
        
        // 임시로 deployer를 proposer로 설정 (나중에 governor로 교체)
        proposers[0] = deployer;
        executors[0] = address(0); // address(0) = 누구나 실행 가능
        
        timelock = new TimelockController(
            2 days,     // minDelay
            proposers,
            executors,
            deployer    // admin
        );
        
        // 3. 거버넌스 컨트랙트 배포
        governor = new MyDAOGovernor(token, timelock);
        
        // 4. 역할 설정: Governor가 Proposer & Canceller
        timelock.grantRole(timelock.PROPOSER_ROLE(), address(governor));
        timelock.grantRole(timelock.CANCELLER_ROLE(), address(governor));
        
        // 5. Admin 권한 포기 (완전한 탈중앙화)
        timelock.revokeRole(timelock.DEFAULT_ADMIN_ROLE(), deployer);
        
        return (token, timelock, governor);
    }
}
```

### 제안 생성 및 투표 예제 (테스트 코드)

```solidity
// Foundry 테스트 코드
contract DAOGovernanceTest is Test {
    GovernanceToken token;
    TimelockController timelock;
    MyDAOGovernor governor;
    
    address alice = makeAddr("alice");
    address bob = makeAddr("bob");
    
    function setUp() public {
        // 배포 및 초기화
        token = new GovernanceToken(address(this));
        
        // 토큰 배분
        token.transfer(alice, 200_000 * 1e18);
        token.transfer(bob, 50_000 * 1e18);
        
        // 반드시 자기 자신에게 위임해야 투표권 활성화
        vm.prank(alice);
        token.delegate(alice);
        
        vm.prank(bob);
        token.delegate(bob);
    }
    
    function testProposalLifecycle() public {
        // 제안: 프로토콜 파라미터 변경
        address[] memory targets = new address[](1);
        uint256[] memory values = new uint256[](1);
        bytes[] memory calldatas = new bytes[](1);
        
        targets[0] = address(protocolContract);
        calldatas[0] = abi.encodeWithSignature(
            "setFee(uint256)", 300 // 3% 수수료
        );
        
        string memory description = "Proposal #1: Set protocol fee to 3%";
        
        // Alice가 제안 생성
        vm.prank(alice);
        uint256 proposalId = governor.propose(
            targets, values, calldatas, description
        );
        
        // Voting Delay 경과
        vm.roll(block.number + governor.votingDelay() + 1);
        
        // 투표 (0: Against, 1: For, 2: Abstain)
        vm.prank(alice);
        governor.castVote(proposalId, 1); // For
        
        vm.prank(bob);
        governor.castVoteWithReason(proposalId, 0, "Fee is too high");
        
        // Voting Period 경과
        vm.roll(block.number + governor.votingPeriod() + 1);
        
        assertEq(
            uint256(governor.state(proposalId)),
            uint256(IGovernor.ProposalState.Succeeded)
        );
        
        // 타임락 큐에 등록
        governor.queue(
            targets, values, calldatas,
            keccak256(bytes(description))
        );
        
        // Timelock 지연 경과
        vm.warp(block.timestamp + timelock.getMinDelay() + 1);
        
        // 실행
        governor.execute(
            targets, values, calldatas,
            keccak256(bytes(description))
        );
    }
}
```

## 주의사항 및 트레이드오프

### 1. 투표 참여율 저하(Voter Apathy) 문제

실제 DAO에서 가장 큰 문제 중 하나다. 쿼럼을 너무 높게 설정하면 제안이 통과되지 않고, 너무 낮게 설정하면 소수의 고래(Whale)가 의사결정을 좌우한다.

**대안**: 위임 메커니즘 활성화, 투표 인센티브(veBal 모델), 낙관적 거버넌스(반대 없으면 자동 통과)

### 2. 플래시론(Flash Loan) 공격

제안 생성 또는 투표 시점에 대규모 토큰을 일시적으로 빌려 투표권을 조작하는 공격이다. ERC-20Votes의 **체크포인트 메커니즘**이 이를 방어한다. 반드시 과거 블록의 스냅샷을 기준으로 투표권을 계산해야 한다.
